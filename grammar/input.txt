type Query {
  hero(episode: Episode): Character,
  starship(id: ID!): Starship,
  episode(number: Number!): Episode
}

type Character {
  name: String!,
  appearsIn: [Episode!]!,
  starship: Starship
}

type Starship {
  id: ID!,
  name: String!
}

type Episode {
  number: Number!,
  name: String!
}

query {
  hero {
    name,
    appearsIn{
      number
    },
    starship{
      id,
      name
    }
  }
}
